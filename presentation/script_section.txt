        window.runDemo = function() {
            const codeInput = document.getElementById('codeInput');
            const emptyMsg = document.getElementById('demo-empty');
            const userCode = codeInput.value.trim();
            
            if (!userCode) {
                alert('Por favor ingresa código o selecciona un ejemplo');
                return;
            }

            let riskType = 'generic';
            for(let i = 1; i <= 9; i++) {
                if(userCode.includes(window.codeExamples[i].substring(0, 30))) {
                    riskType = String(i);
                    break;
                }
            }

            emptyMsg.style.display = 'none';
            const details = window.analysisDetails[riskType] || { title: 'Riesgo Genérico', type: 'Análisis Personalizado', cvss: '6.5', severity: 'MEDIA', color: '#06b6d4', findings: 'Análisis personalizado', technical: 'Código ingresado manualmente', impact: 'Requiere revisión', remediation: 'Contactar seguridad' };
